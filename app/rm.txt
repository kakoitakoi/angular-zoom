import { Component, signal } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FFlowModule } from '@foblex/flow';

interface FlowNode {
  id: string;
  position: { x: number; y: number };
}

interface FlowConnection {
  id: string;
  source: string;
  target: string;
}

interface DiagramState {
  nodes: FlowNode[];
  connections: FlowConnection[];
}

@Component({
  selector: 'app-root',
  standalone: true,
  imports: [CommonModule, FFlowModule],
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.css']
})
export class AppComponent {

  diagramState = signal<DiagramState>({
    nodes: [
      { id: 'node1', position: { x: 100, y: 150 } },
      { id: 'node2', position: { x: 450, y: 200 } }
    ],
    connections: []
  });

  // 🔹 Создание связи
  onCreateConnection(event: any) {

    const newConnection: FlowConnection = {
      id: crypto.randomUUID(),
      source: event.sourceId,
      target: event.targetId
    };

    this.diagramState.update(state => ({
      ...state,
      connections: [...state.connections, newConnection]
    }));
  }

  // 🔹 Удаление связи
  removeConnection(id: string) {
    this.diagramState.update(state => ({
      ...state,
      connections: state.connections.filter(c => c.id !== id)
    }));
  }

}

<f-flow
  fDraggable
  fZoom
  (fCreateConnection)="onCreateConnection($event)">

  <!-- НОДЫ -->
  <f-node
    *ngFor="let node of diagramState().nodes"
    [fNodeId]="node.id"
    [fPosition]="node.position">

    <div class="node">
      {{ node.id }}

      <!-- Порты -->
      <div fOutput></div>
      <div fInput></div>

    </div>

  </f-node>

  <!-- СВЯЗИ -->
  <f-connection
    *ngFor="let conn of diagramState().connections"
    [fConnectionId]="conn.id"
    [fSourceId]="conn.source"
    [fTargetId]="conn.target">

    <!-- КНОПКА УДАЛЕНИЯ -->
    <foreignObject
      width="30"
      height="30"
      style="overflow: visible; pointer-events: none;">

      <div
        xmlns="http://www.w3.org/1999/xhtml"
        class="delete-btn"
        (click)="removeConnection(conn.id)">

        ✕
      </div>

    </foreignObject>

  </f-connection>

</f-flow>
